### Price Calculation Logic

Create a utility function `server/priceEngine.ts`:

```typescript
export interface PriceInputs {
  exwPriceYuan: number;
  unitsPerCarton: number;
  cartonLengthCm: number;
  cartonWidthCm: number;
  cartonHeightCm: number;
  categoryDutyPercent: number;
  categoryIgstPercent: number;
  // Global settings
  exchangeRate: number;        // Yuan to INR, default 12.0
  sourcingCommission: number;  // %, default 5
  freightPerCbm: number;       // INR, default 8000
  insurancePercent: number;    // %, default 0.5
  swSurchargePercent: number;  // % of customs duty, default 10 (fixed by govt)
  ourMarkupPercent: number;    // %, default 25
  targetStoreMargin: number;   // %, default 50 (for MRP suggestion)
}

export function calculatePrices(inputs: PriceInputs) {
  const fobYuan = inputs.exwPriceYuan * (1 + inputs.sourcingCommission / 100);
  const fobInr = fobYuan * inputs.exchangeRate;
  
  const cbmPerCarton = (inputs.cartonLengthCm * inputs.cartonWidthCm * inputs.cartonHeightCm) / 1000000;
  const cbmPerUnit = cbmPerCarton / inputs.unitsPerCarton;
  const freightPerUnit = cbmPerUnit * inputs.freightPerCbm;
  
  const insuranceInr = fobInr * (inputs.insurancePercent / 100);
  const cifInr = fobInr + freightPerUnit + insuranceInr;
  
  const customsDuty = cifInr * (inputs.categoryDutyPercent / 100);
  const swSurcharge = customsDuty * (inputs.swSurchargePercent / 100);
  const assessableValue = cifInr + customsDuty + swSurcharge;
  const igst = assessableValue * (inputs.categoryIgstPercent / 100);
  
  const totalLandedCost = assessableValue + igst;
  const storeLandingPrice = totalLandedCost * (1 + inputs.ourMarkupPercent / 100);
  
  // MRP band assignment
  const MRP_BANDS = [29, 49, 79, 99, 129, 149, 199, 249, 299, 399, 499, 599, 799, 999];
  const targetMrp = storeLandingPrice / (1 - inputs.targetStoreMargin / 100);
  
  // Find nearest band that gives at least 35% margin to store
  let suggestedMrp = MRP_BANDS[MRP_BANDS.length - 1]; // default to highest
  for (const band of MRP_BANDS) {
    const margin = (band - storeLandingPrice) / band;
    if (band >= storeLandingPrice && margin >= 0.35) {
      suggestedMrp = band;
      break;
    }
  }
  
  const storeMarginRs = suggestedMrp - storeLandingPrice;
  const storeMarginPercent = (storeMarginRs / suggestedMrp) * 100;
  
  return {
    fobPriceYuan: round2(fobYuan),
    fobPriceInr: round2(fobInr),
    cbmPerUnit: round4(cbmPerUnit),
    freightPerUnit: round2(freightPerUnit),
    cifPriceInr: round2(cifInr),
    customsDuty: round2(customsDuty),
    swSurcharge: round2(swSurcharge),
    igst: round2(igst),
    totalLandedCost: round2(totalLandedCost),
    storeLandingPrice: round2(storeLandingPrice),
    suggestedMrp,
    storeMarginPercent: round1(storeMarginPercent),
    storeMarginRs: round2(storeMarginRs),
    marginWarning: storeMarginPercent < 35, // Flag if margin too low
  };
}

function round2(n: number) { return Math.round(n * 100) / 100; }
function round4(n: number) { return Math.round(n * 10000) / 10000; }
function round1(n: number) { return Math.round(n * 10) / 10; }
```

## UPGRADE 2: REAL AUTH (HIGH PRIORITY)

### Problem
Login currently fakes auth â€” just redirects. Client dashboard is hardcoded to client ID=1. Need real sessions.

### Implementation

Use Supabase Auth (already have supabase client). Keep it simple:

1. **Register:** Email + password. Creates Supabase auth user + row in `users` table.
2. **Login:** Email + password via Supabase Auth. Returns session token.
3. **Session:** Store token in localStorage. Pass as Bearer token to API.
4. **Middleware:** Express middleware checks auth token on protected routes.
5. **Role check:** Users have `role` field (admin/client). Admin routes check role=admin.
6. **Client mapping:** When client logs in, look up `clients` table by `user_id` to get their client record.

### Auth Flow

```
Login â†’ Supabase Auth â†’ Get user â†’ Check role
  If admin â†’ redirect /admin/dashboard
  If client â†’ look up clients.user_id â†’ redirect /client/dashboard
  
All /api/clients/* calls â†’ use logged-in user's client_id (NOT hardcoded 1)
All /api/admin/* calls â†’ check role=admin
```
