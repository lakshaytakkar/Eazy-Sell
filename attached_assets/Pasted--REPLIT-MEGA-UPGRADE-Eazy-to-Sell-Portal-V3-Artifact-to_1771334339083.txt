**Price Engine** (`server/priceEngine.ts`):

```typescript
export interface GlobalPriceSettings {
  exchangeRate: number;          // Default: 12.0 (¬• to ‚Çπ)
  sourcingCommissionPercent: number;  // Default: 5%
  freightPerCbmInr: number;     // Default: 8000
  insurancePercent: number;      // Default: 0.5%
  swSurchargePercent: number;    // Default: 10% (govt fixed)
  ourMarkupPercent: number;      // Default: 25%
  targetStoreMarginPercent: number;  // Default: 50%
}

export function calculatePrices(
  exwYuan: number,
  unitsPerCarton: number,
  cartonL: number, cartonW: number, cartonH: number,
  categoryDutyPercent: number,
  categoryIgstPercent: number,
  settings: GlobalPriceSettings
) {
  const fobYuan = exwYuan * (1 + settings.sourcingCommissionPercent / 100);
  const fobInr = fobYuan * settings.exchangeRate;
  
  const cbmPerCarton = (cartonL * cartonW * cartonH) / 1_000_000;
  const cbmPerUnit = cbmPerCarton / unitsPerCarton;
  const freightPerUnit = cbmPerUnit * settings.freightPerCbmInr;
  
  const insuranceInr = fobInr * (settings.insurancePercent / 100);
  const cifInr = fobInr + freightPerUnit + insuranceInr;
  
  const customsDuty = cifInr * (categoryDutyPercent / 100);
  const swSurcharge = customsDuty * (settings.swSurchargePercent / 100);
  const igst = (cifInr + customsDuty + swSurcharge) * (categoryIgstPercent / 100);
  
  const totalLandedCost = cifInr + customsDuty + swSurcharge + igst;
  const storeLandingPrice = totalLandedCost * (1 + settings.ourMarkupPercent / 100);
  
  // Snap to nearest MRP band
  const rawMrp = storeLandingPrice / (1 - settings.targetStoreMarginPercent / 100);
  const suggestedMrp = snapToMrpBand(rawMrp);
  
  const storeMarginRs = suggestedMrp - storeLandingPrice;
  const storeMarginPercent = (storeMarginRs / suggestedMrp) * 100;
  
  // Assign price band
  const priceBand = suggestedMrp < 100 ? "10-99" 
    : suggestedMrp < 200 ? "100-199"
    : suggestedMrp < 500 ? "200-499" 
    : "500-999";

  return {
    fobPriceYuan: round2(fobYuan),
    fobPriceInr: round2(fobInr),
    cbmPerUnit: round4(cbmPerUnit),
    freightPerUnit: round2(freightPerUnit),
    cifPriceInr: round2(cifInr),
    customsDuty: round2(customsDuty),
    swSurcharge: round2(swSurcharge),
    igst: round2(igst),
    totalLandedCost: round2(totalLandedCost),
    storeLandingPrice: round2(storeLandingPrice),
    suggestedMrp: round2(suggestedMrp),
    storeMarginPercent: round1(storeMarginPercent),
    storeMarginRs: round2(storeMarginRs),
    priceBand
  };
}

function snapToMrpBand(raw: number): number {
  const bands = [29, 39, 49, 59, 69, 79, 99, 129, 149, 179, 199, 249, 299, 349, 399, 449, 499, 599, 699, 799, 899, 999];
  return bands.find(b => b >= raw) || Math.ceil(raw / 10) * 10 - 1;
}

function round2(n: number) { return Math.round(n * 100) / 100; }
function round4(n: number) { return Math.round(n * 10000) / 10000; }
function round1(n: number) { return Math.round(n * 10) / 10; }
```
